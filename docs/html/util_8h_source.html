104 template <typename... Args>
111 static inline std::string LogPrintStr(const char* fmt, const Args&... args)
112 {
113  std::string log_msg;
114 
115  // Conditionally add timestamp prefix based on g_log_timestamps flag.
116  if (g_log_timestamps.load(std::memory_order_relaxed)) { // Check the flag
117  log_msg = FormatISO8601DateTime(GetUnixEpochTime()) + " ";
118  }
119 
120  try {
121  log_msg += tfm::format(fmt, args...);
122  } catch (tinyformat::format_error& fmterr) {
123  /* Original format string will have newline so don't add one here */
124  log_msg += "Error \"" + std::string(fmterr.what()) + "\" while formatting log message: " + fmt;
125  }
126 
127  log_msg += "\n";
128 
129  return log_msg;
130 }
